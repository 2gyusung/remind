        // 그렇지 않으면 다시 입력 값을 받도록 처리하시오.
        {
            function site() {
                const webSite = prompt('이상 , 이하를 입력하시오', '이하')
                let resultSite = null;
                switch (webSite) {
                    case '이상':
                        resultSite = 'http://naver.com';
                        break;
                    case '이하':
                        resultSite = 'http://google.com';
                        break;
                    default:
                        alert("어떤 값인지 파악이 되지 않습니다.");
                }
                location.href = resultSite;
            }
            site()
        }
